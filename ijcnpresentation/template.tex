% ── Override Dove's black structure with TUD Brilliantblau ────────────────────
% Dove sets everything via `structure`, so this one line cascades to
% titlelike, frametitle, section/subsection in nav bar, block titles, items
\setbeamercolor{structure}{fg=Brilliantblau}

% Title page
\setbeamercolor{title}             {fg=Brilliantblau}
\setbeamercolor{subtitle}          {fg=Brilliantblau}
\setbeamercolor{author}            {fg=Brilliantblau}
\setbeamercolor{institute}         {fg=Brilliantblau}
\setbeamercolor{date}              {fg=Brilliantblau}

% Frame titles
\setbeamercolor{frametitle}        {fg=Brilliantblau, bg=white}
\setbeamercolor{framesubtitle}     {fg=Grau80}

% Nav bar (Berlin mini-frames) — Dove leaves these white/black, give them TUD tones
\setbeamercolor{palette primary}   {fg=Brilliantblau}
\setbeamercolor{palette secondary} {fg=Dunkelblau}
\setbeamercolor{palette tertiary}  {fg=Blau1}
\setbeamercolor{palette quaternary}{fg=Brill}
% Section/subsection labels in the nav bar inherit from palette — already set above
\setbeamercolor{section in head/foot}   {fg=Brilliantblau}
\setbeamercolor{subsection in head/foot}{fg=Dunkelblau}

% Blocks — Dove inherits from structure/alerted/example text; refine here
\setbeamercolor{block title}         {fg=Brilliantblau, bg=Grau20}
\setbeamercolor{block body}          {fg=Schwarz,       bg=Grau10}
\setbeamercolor{block title example} {fg=Weiss,         bg=Violett1}
\setbeamercolor{block body example}  {fg=Schwarz,       bg=Violett2}
\setbeamercolor{block title alerted} {fg=Weiss,         bg=Rot1}
\setbeamercolor{block body alerted}  {fg=Schwarz,       bg=Rot2}

\setbeamercolor{alerted text}        {fg=Rot1}
\setbeamercolor{item projected}      {fg=white, bg=Brilliantblau}

\setbeamercolor{itemize item}        {fg=Brilliantblau}
\setbeamercolor{itemize subitem}     {fg=Blau1}
\setbeamercolor{itemize subsubitem}  {fg=Blau2}
 
\setbeamertemplate{itemize item}     {\raise1.25pt\hbox{\donotcoloroutermaths$\blacktriangleright$}}
\setbeamertemplate{itemize subitem}  {\raise1.25pt\hbox{\donotcoloroutermaths$\blacktriangleright$}}
\setbeamertemplate{itemize subsubitem}{\raise1.25pt\hbox{\donotcoloroutermaths$\blacktriangleright$}}
 
% ── Blue triangles for enumerate and TOC too ──────────────────────────────────
\setbeamercolor{item projected}      {fg=white, bg=Brilliantblau}
\setbeamertemplate{enumerate item}   {\usebeamercolor[bg]{item projected}\raise1.25pt\hbox{\donotcoloroutermaths$\blacktriangleright$}~\insertenumlabel}
 
\setbeamercolor{section in toc}      {fg=Brilliantblau}
\setbeamercolor{subsection in toc}   {fg=Blau1}
\setbeamertemplate{sections/subsections in toc}[ball]
\setbeamertemplate{section in toc}{%
  {\color{Brilliantblau}$\blacktriangleright$}~\inserttocsection}
\setbeamertemplate{subsection in toc}{%
  \hspace{1em}{\color{Blau1}$\blacktriangleright$}~\inserttocsubsection\par}
